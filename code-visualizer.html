示例代码

滤波器设计示例

import numpy as np
import matplotlib.pyplot as plt
from scipy.signal import butter, filtfilt

# 生成测试信号
t = np.linspace(0, 1, 1000)
signal = np.sin(2*np.pi*5*t) + 0.5*np.sin(2*np.pi*50*t)
noise = 0.3*np.random.randn(len(t))
noisy_signal = signal + noise

# 设计低通滤波器
fs = 1000
fc = 15
b, a = butter(4, fc/(fs/2), btype='low')
filtered_signal = filtfilt(b, a, noisy_signal)

# 绘图对比
plt.figure(figsize=(10, 6))
plt.subplot(2,1,1)
plt.plot(t[:200], noisy_signal[:200], label='含噪信号')
plt.legend()
plt.title('滤波前')

plt.subplot(2,1,2)
plt.plot(t[:200], filtered_signal[:200], label='滤波后')
plt.legend()
plt.title('滤波后')
plt.tight_layout()
plt.show()

频谱分析示例

import numpy as np
import matplotlib.pyplot as plt

# 生成复合信号
fs = 1000
t = np.linspace(0, 2, 2*fs)
f1, f2, f3 = 10, 50, 120
signal = np.sin(2*np.pi*f1*t) + 0.5*np.sin(2*np.pi*f2*t) + 0.3*np.sin(2*np.pi*f3*t)

# FFT分析
fft_result = np.fft.fft(signal)
freqs = np.fft.fftfreq(len(signal), 1/fs)

# 绘制频谱
plt.figure(figsize=(10, 4))
plt.plot(freqs[:len(freqs)//2], np.abs(fft_result[:len(fft_result)//2]))
plt.title('信号频谱分析')
plt.xlabel('频率 (Hz)')
plt.ylabel('幅度')
plt.grid(True)
plt.show()
